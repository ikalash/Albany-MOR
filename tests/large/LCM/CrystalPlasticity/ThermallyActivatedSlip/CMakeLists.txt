
# Create a symbolic link for exodiff
execute_process(COMMAND ${CMAKE_COMMAND} -E create_symlink
                ${SEACAS_EXODIFF} ${CMAKE_CURRENT_BINARY_DIR}/exodiff)

# Copy Input files from source to binary dir
execute_process(COMMAND ${CMAKE_COMMAND} -E create_symlink
  ${runtest.cmake} ${CMAKE_CURRENT_BINARY_DIR}/runtest.cmake)

configure_file(${CMAKE_CURRENT_SOURCE_DIR}/ThermallyActivatedSlip.g
               ${CMAKE_CURRENT_BINARY_DIR}/ThermallyActivatedSlip.g COPYONLY)

configure_file(${CMAKE_CURRENT_SOURCE_DIR}/ThermallyActivatedSlip.yaml
               ${CMAKE_CURRENT_BINARY_DIR}/ThermallyActivatedSlip.yaml COPYONLY)
configure_file(${CMAKE_CURRENT_SOURCE_DIR}/ThermallyActivatedSlip_Material.yaml
               ${CMAKE_CURRENT_BINARY_DIR}/ThermallyActivatedSlip_Material.yaml COPYONLY)
configure_file(${CMAKE_CURRENT_SOURCE_DIR}/ThermallyActivatedSlip.exodiff
               ${CMAKE_CURRENT_BINARY_DIR}/ThermallyActivatedSlip.exodiff COPYONLY)
configure_file(${CMAKE_CURRENT_SOURCE_DIR}/ThermallyActivatedSlip.gold.exo
               ${CMAKE_CURRENT_BINARY_DIR}/ThermallyActivatedSlip.gold.exo COPYONLY)

# Name the test with the directory name
get_filename_component(testName ${CMAKE_CURRENT_SOURCE_DIR} NAME)
# Add the tests
if(ALBANY_IFPACK2)
 SET(OUTFILE "ThermallyActivatedSlip.exo")
 SET(REF_FILE "ThermallyActivatedSlip.gold.exo")
 add_test(NAME CrystalPlasticity_${testName}
          COMMAND ${CMAKE_COMMAND} "-DTEST_PROG=${SerialAlbanyT.exe}"
          -DTEST_NAME=ThermallyActivatedSlip -DTEST_ARGS=ThermallyActivatedSlip.yaml -DMPIMNP=1
          -DSEACAS_EXODIFF=${SEACAS_EXODIFF}
          -DREF_FILENAME=${REF_FILE} -DOUTPUT_FILENAME=${OUTFILE}
          -DDATA_DIR=${CMAKE_CURRENT_SOURCE_DIR} -P ${runtest.cmake})
endif()
